Step 3: Define Tools

@tool
def search_financial_news(ticker: str) -> str:
    """Search for recent financial news about a stock ticker.

    Args:
        ticker: Stock ticker symbol (e.g., NVDA, AAPL, MSFT)
    """
    ticker = ticker.upper().strip()
    if ticker not in NEWS_DATABASE:
        return f"No news found for ticker: {ticker}. Available: {', '.join(NEWS_DATABASE.keys())}"

    articles = NEWS_DATABASE[ticker]
    result = f"## Recent News for {ticker}\n\n"
    for i, article in enumerate(articles, 1):
        result += f"### {i}. {article['title']}\n"
        result += f"**Source:** {article['source']} | **Sentiment:** {article['sentiment']}\n"
        result += f"{article['summary']}\n\n"

    return result


@tool
def get_stock_data(ticker: str) -> str:
    """Get current stock data including price, P/E ratio, market cap, and other metrics.

    Args:
        ticker: Stock ticker symbol (e.g., NVDA, AAPL, MSFT)
    """
    ticker = ticker.upper().strip()
    if ticker not in STOCK_DATABASE:
        return f"Unknown ticker: {ticker}. Available: {', '.join(STOCK_DATABASE.keys())}"

    stock = STOCK_DATABASE[ticker]
    result = f"## Stock Data: {stock['name']} ({ticker})\n\n"
    result += f"| Metric | Value |\n|--------|-------|\n"
    result += f"| Price | ${stock['price']:.2f} |\n"
    result += f"| Change | {stock['change_pct']:+.1f}% |\n"
    result += f"| P/E Ratio | {stock['pe_ratio']} |\n"
    result += f"| Forward P/E | {stock['forward_pe']} |\n"
    result += f"| Market Cap | {stock['market_cap']} |\n"
    result += f"| EPS | ${stock['eps']:.2f} |\n"
    result += f"| 52-Week High | ${stock['52w_high']:.2f} |\n"
    result += f"| 52-Week Low | ${stock['52w_low']:.2f} |\n"
    result += f"| Volume | {stock['volume']} |\n"
    result += f"| Revenue Growth | {stock['revenue_growth']}% |\n"
    result += f"| Profit Margin | {stock['profit_margin']}% |\n"
    result += f"| Debt/Equity | {stock['debt_to_equity']} |\n"
    result += f"| Dividend Yield | {stock['dividend_yield']}% |\n"
    result += f"| Beta | {stock['beta']} |\n"
    result += f"| Analyst Rating | {stock['analyst_rating']} |\n"
    result += f"| Price Target | ${stock['price_target']:.2f} |\n"
    result += f"| Sector | {stock['sector']} |\n"
    result += f"| Industry | {stock['industry']} |\n"

    return result


@tool
def calculate_financial_metrics(tickers: str) -> str:
    """Calculate comparative financial metrics for a comma-separated list of tickers.

    Args:
        tickers: Comma-separated list of ticker symbols (e.g., "NVDA,AAPL,MSFT")
    """
    ticker_list = [t.strip().upper() for t in tickers.split(",")]
    valid_tickers = [t for t in ticker_list if t in STOCK_DATABASE]

    if not valid_tickers:
        return "No valid tickers provided."

    result = "## Comparative Financial Metrics\n\n"

    # Comparison table
    header = "| Metric | " + " | ".join(valid_tickers) + " |\n"
    separator = "|--------|" + "|".join(["--------"] * len(valid_tickers)) + "|\n"
    result += header + separator

    metrics = [
        ("Price", lambda s: f"${s['price']:.2f}"),
        ("P/E Ratio", lambda s: f"{s['pe_ratio']}"),
        ("Forward P/E", lambda s: f"{s['forward_pe']}"),
        ("Market Cap", lambda s: f"{s['market_cap']}"),
        ("Rev Growth", lambda s: f"{s['revenue_growth']}%"),
        ("Profit Margin", lambda s: f"{s['profit_margin']}%"),
        ("Debt/Equity", lambda s: f"{s['debt_to_equity']}"),
        ("Beta", lambda s: f"{s['beta']}"),
        ("Analyst", lambda s: f"{s['analyst_rating']}"),
        ("Upside", lambda s: f"{((s['price_target'] - s['price']) / s['price'] * 100):.1f}%"),
    ]

    for name, fn in metrics:
        values = " | ".join(fn(STOCK_DATABASE[t]) for t in valid_tickers)
        result += f"| {name} | {values} |\n"

    # Rankings
    result += "\n### Rankings\n\n"
    stocks = [(t, STOCK_DATABASE[t]) for t in valid_tickers]

    result += "**By Revenue Growth:** "
    result += ", ".join(
        f"{t} ({s['revenue_growth']}%)"
        for t, s in sorted(stocks, key=lambda x: x[1]["revenue_growth"], reverse=True)
    ) + "\n\n"

    result += "**By Profit Margin:** "
    result += ", ".join(
        f"{t} ({s['profit_margin']}%)"
        for t, s in sorted(stocks, key=lambda x: x[1]["profit_margin"], reverse=True)
    ) + "\n\n"

    result += "**By Value (Lowest P/E):** "
    result += ", ".join(
        f"{t} ({s['pe_ratio']})"
        for t, s in sorted(stocks, key=lambda x: x[1]["pe_ratio"])
    ) + "\n\n"

    result += "**By Risk (Beta):** "
    result += ", ".join(
        f"{t} ({s['beta']})"
        for t, s in sorted(stocks, key=lambda x: x[1]["beta"])
    ) + "\n"

    return result

import sqlite3

DB_PATH = "analytics.db"


def setup_database():
    """Create and populate the sample analytics database."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    # ─── Create Tables ───

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS products (
            id INTEGER PRIMARY KEY,
            name TEXT NOT NULL,
            category TEXT NOT NULL,
            price REAL NOT NULL,
            cost REAL NOT NULL,
            stock_quantity INTEGER NOT NULL,
            supplier TEXT NOT NULL,
            created_date TEXT NOT NULL
        )
    """)

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS customers (
            id INTEGER PRIMARY KEY,
            name TEXT NOT NULL,
            email TEXT NOT NULL,
            region TEXT NOT NULL,

# ─── Tools ───

@tool
def explore_schema() -> str:
    """Explore the database schema. Returns all table names, columns, types,
    and sample data. Call this FIRST before writing any SQL queries."""
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()

    # Get all tables
    cursor.execute("SELECT name FROM sqlite_master WHERE type='table'")
    tables = [row[0] for row in cursor.fetchall()]

    result = "## Database Schema\n\n"

    for table in tables:
        result += f"### Table: {table}\n"

        # Get column info
        cursor.execute(f"PRAGMA table_info({table})")
        columns = cursor.fetchall()
        result += "| Column | Type | Nullable | PK |\n"
        result += "|--------|------|----------|----|\n"
        for col in columns:
            result += f"| {col[1]} | {col[2]} | {'YES' if not col[3] else 'NO'} | {'YES' if col[5] else 'NO'} |\n"

        # Row count
        cursor.execute(f"SELECT COUNT(*) FROM {table}")
        count = cursor.fetchone()[0]
        result += f"\nRow count: {count}\n"

        # Sample rows
        cursor.execute(f"SELECT * FROM {table} LIMIT 3")
        samples = cursor.fetchall()
        col_names = [desc[0] for desc in cursor.description]
        result += f"\nSample data:\n"
        result += "| " + " | ".join(col_names) + " |\n"
        result += "|" + "|".join(["---"] * len(col_names)) + "|\n"
        for row in samples:
            result += "| " + " | ".join(str(v) for v in row) + " |\n"

        result += "\n"

    conn.close()
    return result


@tool
def run_sql_query(query: str) -> str:
    """Run a SQL query against the analytics database.
    SAFETY: Only SELECT statements are allowed. No INSERT, UPDATE, DELETE, DROP, etc.

    Args:
        query: A SQL SELECT query to execute
    """
    # ─── Safety Checks ───
    query_upper = query.upper().strip()

    # Block dangerous operations
    blocked_keywords = ["INSERT", "UPDATE", "DELETE", "DROP", "ALTER",
                        "CREATE", "TRUNCATE", "REPLACE", "EXEC", "GRANT",
                        "REVOKE", "ATTACH", "DETACH", "PRAGMA"]

    for keyword in blocked_keywords:
        if keyword in query_upper:
            return f"ERROR: {keyword} statements are not allowed. Only SELECT queries permitted."

    if not query_upper.startswith("SELECT"):
        return "ERROR: Only SELECT queries are allowed."

    # ─── Execute Query ───
    try:
        conn = sqlite3.connect(DB_PATH)
        cursor = conn.cursor()
        cursor.execute(query)

        columns = [desc[0] for desc in cursor.description]
        rows = cursor.fetchall()
        conn.close()

        if not rows:
            return "Query returned no results."

        # Format as markdown table
        result = f"Query returned {len(rows)} rows:\n\n"
        result += "| " + " | ".join(columns) + " |\n"
        result += "|" + "|".join(["---"] * len(columns)) + "|\n"
        for row in rows[:50]:  # Limit to 50 rows
            result += "| " + " | ".join(str(v) for v in row) + " |\n"

        if len(rows) > 50:
            result += f"\n... and {len(rows) - 50} more rows (showing first 50)"

        return result

    except Exception as e:
        return f"SQL Error: {e}\n\nPlease check your query syntax and try again."

- Start with a brief restatement of the question
- Show the SQL query you're running (in a code block)
- Present results in a clear table or bullet list
- Add a brief insight or observation about the data

- NEVER modify data — only SELECT queries
- NEVER expose raw connection strings or credentials

Key Takeaways

  1. Structure beats intelligence. A well-structured agent with clear system prompts outperforms a more powerful model with vague instructions.
  2. Decomposition is everything. Break complex tasks into sub-agents with focused roles. Each agent should do one thing well.
  3. The filesystem is your memory. Use file-based handoff to preserve context across agents. Never rely on a single conversation thread for complex work.
  4. Tools are the bridge. The @tool decorator lets you connect LLMs to any API, database, or service. The docstring is the interface.
  5. Safety first. Always validate inputs, restrict operations (SELECT-only for SQL), and add guardrails to your tools.
  6. Start simple, evolve. Begin with Pattern 1 (simple agent) and add complexity only when needed. Not every problem requires a multi-agent system.
